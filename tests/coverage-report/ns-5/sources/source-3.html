323      * @param pX x coordinate from the point
324      * @param pY y coordinate from the point
325      * @param s1X x coordinate from the first point of the segment
326      * @param s1Y y coordinate from the first point of the segment

329      * @return the minimal distance
330      */
331     public static double getDistanceFromPointToSegment(int pX, int pY, int s1X, int s1Y, int s2X, int s2Y){
332        // héron's formula :
333         double distps1 = dist(pX, pY, s1X, s1Y);
334         double distps2 = dist(pX, pY, s2X, s2Y);
335         double dists1s2 = dist(s1X, s1Y, s2X, s2Y);
336         double p = (distps1 + distps2 + dists1s2)/2; // half perimeter
337         double A = Math.sqrt(p*(p - distps1)*(p-distps2)*(p-dists1s2)); // heron's formula
338         double hauteur = (A * 2)/dists1s2; //area of a triangle
339 
340         //check if the point is on the segment (if their is an angle more than 90° in the triangle
341         // al-khashi :
342         double alpha = Math.acos((distps2*distps2 - dists1s2*dists1s2 - distps1*distps1)/(-2*dists1s2*distps1));
343         double gamma = Math.acos((distps1*distps1 - dists1s2*dists1s2 - distps2*distps2)/(-2*dists1s2*distps2));
344         if(alpha > Math.PI/2 || gamma > Math.PI/2) {
345             return Math.min(distps1, distps2);
346         }else{
347             return hauteur;
348         }
