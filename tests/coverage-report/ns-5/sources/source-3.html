327      * @param pX x coordinate from the point
328      * @param pY y coordinate from the point
329      * @param s1X x coordinate from the first point of the segment
330      * @param s1Y y coordinate from the first point of the segment
331      * @param s2X x coordinate from the second point of the segment
332      * @param s2Y y coordinate from the second point of the segment
333      * @return the minimal distance
334      */
335     public static double getDistanceFromPointToSegment(int pX, int pY, int s1X, int s1Y, int s2X, int s2Y){
336        // héron's formula :
337         double distps1 = dist(pX, pY, s1X, s1Y);
338         double distps2 = dist(pX, pY, s2X, s2Y);
339         double dists1s2 = dist(s1X, s1Y, s2X, s2Y);
340         double p = (distps1 + distps2 + dists1s2)/2; // half perimeter
341         double A = Math.sqrt(p*(p - distps1)*(p-distps2)*(p-dists1s2)); // heron's formula
342         double hauteur = (A * 2)/dists1s2; //area of a triangle
343 
344         return Math.min(Math.min(distps1, distps2), hauteur);
345     }
